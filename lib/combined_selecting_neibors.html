Standard: larger than threshold and max n is 5.

aaa<- function(x){
  a1<- x[order(-x)][1:n]*(x[order(-x)][1:n]>=threshold)
  a1[a1==0]<- NA
  res<- which(x %in% a1)[1:n]
  return(res)
}
t(apply(data, 1, function(x) aaa(x)))
